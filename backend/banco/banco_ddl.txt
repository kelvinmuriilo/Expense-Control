-- TABELA TIPOS

CREATE TABLE TIPOS(
	idTipo int IDENTITY(1,1) NOT NULL,
	descricao varchar(200) NOT NULL,
	CONSTRAINT PK_TIPOS PRIMARY KEY (idTipo)
);


-- TABELA DESPESAS

CREATE TABLE DESPESAS(
    idDespesa int IDENTITY(1,1) NOT NULL,
    dataCadastro date NOT NULL,
    descricao varchar(200) NOT NULL,
    valor Decimal(10,3) NOT NULL,
    idTipo int NOT NULL,

    CONSTRAINT PK_DESPESAS PRIMARY KEY (idDespesa),
    CONSTRAINT FK_DESPESAS_idTipo FOREIGN KEY (idTipo)
    REFERENCES TIPOS (idTipo)
);

SELECT * FROM TIPOS;
